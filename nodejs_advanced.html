Node.js 심화 (Advanced)

단순한 API 서버 구축을 넘어, Node.js의 한계를 돌파합니다. 비동기 이벤트 루프의 심연, 메모리 폭발을 막는 스트림, 그리고 실시간 통신의 세계로 들어갑니다.

1. 이벤트 루프(Event Loop)의 진짜 우선순위

Node.js는 싱글 스레드지만 libuv라는 라이브러리 덕분에 논블로킹(Non-blocking) I/O가 가능합니다. 비동기 함수들이 어떤 순서로 실행되는지 정확히 아는 것이 성능 최적화의 열쇠입니다.

const fs = require('fs'); console.log('1. 메인 코드 실행 (Call Stack)'); // 타이머 페이즈 (Timer Phase) setTimeout(() => { console.log('4. setTimeout 실행'); }, 0); // 체크 페이즈 (Check Phase) setImmediate(() => { console.log('5. setImmediate 실행'); }); // 마이크로태스크 큐 (최우선순위 VIP) Promise.resolve().then(() => { console.log('3. Promise.then 실행'); }); // 마이크로태스크 중에서도 초특급 VIP process.nextTick(() => { console.log('2. process.nextTick 실행'); });
> Terminal - node event-loop.js
1. 메인 코드 실행 (Call Stack)
2. process.nextTick 실행
3. Promise.then 실행
4. setTimeout 실행
5. setImmediate 실행
💡 실행 순서의 비밀 모든 동기 코드가 먼저 실행된 후, 비동기 작업 중 마이크로태스크 큐(nextTick, Promise)가 가장 먼저 새치기를 해서 실행됩니다. 그 후 이벤트 루프의 각 페이즈(Timer -> Poll -> Check)를 순환하며 실행됩니다.

2. 메모리 폭발을 막아라: Streams & Buffers

만약 메모리(RAM)가 100MB인 서버에서 1GB짜리 동영상 파일을 읽어서 사용자에게 보낸다면 어떻게 될까요? 파일을 한 번에 다 읽는 fs.readFile을 쓰면 서버가 터집니다. 이때 **스트림(Stream)**을 사용해야 합니다.

잘게 쪼개서 수도관으로 흘려보내기 (pipe)

파일을 작은 버퍼(Buffer) 단위로 잘라서 조각조각 전송합니다. 유튜브 동영상이 로딩되면서 재생되는 것과 같은 원리입니다.

const fs = require('fs'); const server = require('http').createServer(); server.on('request', (req, res) => { // ❌ 하수들의 방식 (메모리 터짐) /* fs.readFile('./big-video.mp4', (err, data) => { res.end(data); }); */ // 🟢 고수들의 방식 (스트림 파이프라인) // 파일을 조금씩 읽어오는 수도꼭지(ReadStream)를 만듭니다. const readStream = fs.createReadStream('./big-video.mp4'); // 읽어온 데이터를 바로 클라이언트 응답(res) 수도관으로 연결(pipe)합니다. readStream.pipe(res); }); server.listen(8000);

3. 양방향 실시간 통신 (WebSockets & Socket.io)

일반적인 HTTP 통신은 클라이언트가 요청(Req)해야만 서버가 응답(Res)하고 연결을 끊어버립니다. 채팅 앱이나 주식 차트처럼 서버가 원할 때 클라이언트에게 데이터를 쏴주려면 웹소켓(WebSocket)이 필요합니다.

const express = require('express'); const http = require('http'); // npm install socket.io 필수! const { Server } = require('socket.io'); const app = express(); const server = http.createServer(app); const io = new Server(server); // 누군가 웹소켓에 '연결'되었을 때 io.on('connection', (socket) => { console.log('유저가 접속했습니다. ID: ' + socket.id); // 클라이언트가 'chat message'라는 이름으로 데이터를 보내면 socket.on('chat message', (msg) => { // 접속한 '모든' 유저에게 이 메시지를 뿌립니다 (브로드캐스트) io.emit('chat message', msg); }); }); server.listen(3000);

4. 클러스터(Cluster) 모듈과 CPU 100% 활용

Node.js는 기본적으로 스레드(코어)를 단 1개만 사용합니다. 내 컴퓨터 코어가 8개라면 나머지 7개는 놀고 있다는 뜻입니다. 클러스터 모듈을 사용하면 CPU 코어 수만큼 Node.js 프로세스를 복제(Fork)하여 병렬로 일을 처리할 수 있습니다.

워커(Worker) 프로세스 생성

실무에서는 이 과정을 직접 코딩하기보다 PM2라는 프로세스 매니저 도구를 사용해 pm2 start app.js -i max 한 줄로 해결합니다.

const cluster = require('cluster'); const os = require('os'); if (cluster.isMaster) { // 1. 마스터(본체) 프로세스일 때: CPU 코어 개수를 파악합니다. const numCPUs = os.cpus().length; console.log(`마스터 프로세스 작동. 총 코어 수: ` + numCPUs); // 2. 코어 개수만큼 워커(분신)를 생성(Fork)합니다. for (let i = 0; i < numCPUs; i++) { cluster.fork(); } // 분신 하나가 죽으면 새로운 분신을 즉시 다시 만듭니다. (자동 복구) cluster.on('exit', (worker) => { console.log(`워커 ${worker.process.pid}가 죽었습니다. 재시작합니다...`); cluster.fork(); }); } else { // 3. 분신(Worker)들은 각자 웹 서버를 실행하여 트래픽을 나눠 받습니다. const express = require('express'); const app = express(); app.listen(3000, () => { console.log(`워커 ${process.pid} 작동 중...`); }); }

5. 보안과 인증: JWT 미들웨어

사용자가 로그인 후 발행받은 JWT(토큰)를 요청 헤더에 담아 보낼 때, Express 라우터에 도달하기 전 미들웨어(Middleware)에서 이 토큰이 진짜인지 위조되었는지 검사하는 로직입니다.

const jwt = require('jsonwebtoken'); // 토큰을 검사하는 커스텀 미들웨어 함수 const verifyToken = (req, res, next) => { // 프론트엔드가 요청 헤더(Authorization)에 담아 보낸 토큰 추출 const authHeader = req.headers.authorization; if (authHeader) { const token = authHeader.split(' ')[1]; // "Bearer 토큰값" 에서 값만 분리 // 시크릿 키를 이용해 토큰 위조/만료 여부 검사 jwt.verify(token, process.env.JWT_SECRET, (err, user) => { if (err) return res.status(403).json({ msg: "토큰이 유효하지 않습니다." }); req.user = user; // 검증 통과! 유저 정보를 req 객체에 담음 next(); // 다음 라우터로 통과시켜줌! (가장 중요) }); } else { res.status(401).json({ msg: "인증 토큰이 없습니다." }); } }; // 마이페이지 API에 미들웨어(verifyToken) 장착! app.get('/api/mypage', verifyToken, (req, res) => { // 미들웨어를 무사히 통과한 유저만 이 코드를 실행할 수 있습니다. res.json({ msg: "비밀 정보입니다.", userInfo: req.user }); });
